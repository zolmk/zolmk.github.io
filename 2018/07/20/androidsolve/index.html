RecyclerView的运用

如何给Item中的组件添加监听呐？
尝试了很多地方，在网上没有找到解决方法，如果直接添加到Item.OnClickListener中，只有当点击两次以上的时候才会触发，不符合要求，最后在OnBidViewHolder中实现了该功能。

//实现RecyclerView列表
class Adapter extends RecyclerView.Adapter<Adapter.VH>{

    static public class VH extends RecyclerView.ViewHolder{
        public final TextView searchLocation;
        public final TextView location;
        public final ImageView searchType;
        public VH(View itemView) {
            super(itemView);
            searchLocation = (TextView) itemView.findViewById(R.id.search_name);
            location = (TextView) itemView.findViewById(R.id.localtion_name);
            searchType = (ImageView) itemView.findViewById(R.id.type_search);
        }
    }
    public Context context;
    private Adapter.OnItemClickListener onItemClickListener;
    private ArrayList<ItemData> mDatas;

    public Adapter(ArrayList<ItemData> data, Context context){
        this.mDatas = data;
        this.context = context;
    }
    @Override
    public VH onCreateViewHolder(ViewGroup parent, int i){
        View v = LayoutInflater.from(parent.getContext()).inflate(R.layout.toursearch_item,parent,false);
        return new VH(v);
    }

    @Override
    public void onBindViewHolder(final VH vh,final int i) {

        vh.searchLocation.setText(mDatas.get(i).getSearchLocation());
        vh.location.setText(mDatas.get(i).getLocation());

        vh.itemView.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                if(onItemClickListener != null){
                    int pos = vh.getLayoutPosition();
                    onItemClickListener.onItemClick(vh.itemView,pos);
                }
            }

        });
        vh.itemView.setOnLongClickListener(new View.OnLongClickListener(){

            @Override
            public boolean onLongClick(View view) {
                if(onItemClickListener != null){
                    int pos = vh.getLayoutPosition();
                    onItemClickListener.onItemLongClick(vh.itemView,pos);
                }
                return true;
            }
        });
        vh.itemView.findViewById(R.id.location).setOnClickListener(new View.OnClickListener(){
            public void onClick(View view) {
                EventBus.getDefault().post(new Message(mDatas.get(i).getWd(),mDatas.get(i).getJd()));
                context.startActivity(new Intent(context,EasyTour.class));
            }
        });

    }

    @Override
    public int getItemCount() {
        return mDatas == null?0:this.mDatas.size();
    }

    //定义点击回调接口
    public interface OnItemClickListener{
        void onItemClick(View v,int postion);
        void onItemLongClick(View v,int postion);
    }

    public void setOnItemClickListener(Adapter.OnItemClickListener listener){
        this.onItemClickListener = listener;
    }

    //改元素

    public void addNewItem(ItemData data){
        if(mDatas == null){
            mDatas = new ArrayList<ItemData>();
        }
        mDatas.add(data);
        notifyItemInserted(mDatas.size());
    }
    public void deleteItem(){
        if(mDatas == null || mDatas.isEmpty()){
            return;
        }
        mDatas.remove(0);
        notifyItemRemoved(0);
    }
    public void updateDate(ArrayList<ItemData> data){
        this.mDatas = data;
        notifyDataSetChanged();
    }
    public double getJdByIndex(int index){
        return mDatas.get(index).getJd();
    }
    public double getWdByIndex(int index){
        return mDatas.get(index).getWd();
    }
}
